package Model;

import Controller.Players;
import View.Table;
import java.util.ArrayList;


public class PokerHandEval {
  Players player;
  Table table;


  public PokerHandEval(Players player, Table table) {
    this.player = player;
    this.table = table;
  }



  // High Card for player & table

  public static Card highCard(ArrayList<Card> hand) {
    Card currHighCard = null;

    for (int i = 0; i < hand.size(); i++) {
      if (hand.get(i).compareTo(currHighCard) > 0) {
        currHighCard = hand.get(i);
      }

    }
    return currHighCard;
  }


  public static Card highCard(Players player, Table table) {
    int card1 = 0;
    int card2 = 1;

    Card currHighCard = null;

    if (player.getHand().getCard(card1).compareTo(player.getHand().getCard(card2)) == 1) {
      currHighCard = player.getHand().getCard(card1);

    } else if (player.getHand().getCard(card1).compareTo(player.getHand().getCard(card2)) == -1) {
      currHighCard = player.getHand().getCard(card2);
    }

    // This will iterate through the board/table cards to find the highest card

    for (int i = 0; i < table.size(); i++) {
      if (table.getBoard().getCard(i).compareTo(currHighCard) > 0) {
        currHighCard = table.getBoard().getCard(i);
      }

    }
    return currHighCard;
  }


  public static Hand pair(Players player, Table table) {
    int card1 = 0;
    int card2 = 1;

    ArrayList<Card> hand = new ArrayList<>();
    Hand pair = new Hand(hand);

    // case 1 pair in hand

    if (player.getHand().getCard(card1).compareTo(player.getHand().getCard(card2)) == 0) {
      return player.getHand();
    }

    if (table.getBoard() != null) {

      // case pair in hand and table

      // case 3 pair on table but not in hand

      for (int playerCard = 0; playerCard < player.getHand().size(); playerCard++)
        for (int tableCard = 0; tableCard < table.size(); tableCard++) {

          if (table.getBoard().getCard(card1).compareTo(player.getHand().getCard(playerCard))
              == 0) {
            pair.addCard(player.getHand().getCard(playerCard));
            pair.addCard(table.getBoard().getCard(card1));
          }
          return pair;

        }

    }
    return pairOnTableHelper(table);

  }


  public static Hand pairOnTableHelper(Table table) {

    ArrayList<Card> hand = new ArrayList<>();
    Hand pair = new Hand(hand);

    for (int i = 0; i < table.size(); i++) {
      for (int j = 1; j < table.size(); j++) {
        if (table.getBoard().getCard(i).compareTo(table.getBoard().getCard(j)) == 0) {
          pair.addCard(table.getBoard().getCard(i));
          pair.addCard(table.getBoard().getCard(j));
        }
      }
      return pair;

    }
    return null;

  }

}
