package Model;


import java.util.ArrayList;
import java.util.Objects;

public class Hand  {

  public ArrayList<Card> hand;



  public Hand(ArrayList<Card> hand) {
    this.hand = hand;

  }



  public Integer size() {
    return this.hand.size();
  }

  /**
   * Getter for a Hand
   *
   * @return ArrayList for a Hand
   */

  public ArrayList<Card> getHand() {
    return this.hand;
  }


  /**
   * Add a Card to the hand
   *
   * @param card, being added to the hand
   */

  public void addCard(Card card) {
    this.hand.add(card);

  }

  /**
   * Getter method for a card in hand.
   *
   * @param cardIndex
   * @return
   */


  public Card getCard(Integer cardIndex) {
    return this.hand.get(cardIndex);

  }



  @Override
  public boolean equals(Object o) {
    if (this == o)
      return true;
    if (o == null || getClass() != o.getClass())
      return false;
    Hand that = (Hand) o;
    return Objects.equals(getHand(), that.getClass());
  }



  @Override
  public int hashCode() {
    return Objects.hash(hand);
  }


  @Override
  public String toString() {
    return "Hand :: " + hand;
  }
}
