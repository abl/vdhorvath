package Model;


import java.util.ArrayList;


public class HighCard extends Hand {

  public HighCard(ArrayList<Card> hand) {
    super(hand);
  }


  public Card highCard(ArrayList<Card> hand) {
    Card currHighCard = null;

    for (int i = 0; i < hand.size(); i++) {
      if (hand.get(i).compareTo(currHighCard) > 0) {
        currHighCard = hand.get(i);
      }

    }
    return currHighCard;
  }





  @Override
  public String toString() {
    return super.toString() + " " + "\nYour HighCard " + " " + highCard(getHand());

  }



}
